77 bool Unpacker::ExtractEvent(Event &event, int &n_data, int i, int64_t &timeOfEvent)
78 {
79  int stop = i;
80  int start = i;
81  int end = -1;
82 
83  // Time of the event.
84  int64_t ClockStart = timeOfEvent;
85 
86  // Finding all words with timestamp >= -35 from the start time.
87  for (int j = i ; j >= 1 ; --j){
88  __builtin_prefetch(&buffer[j-1]);
89  __builtin_prefetch(&buffer[j-2]);
90  start = j;
91  int64_t timediff = ClockStart - (*buffer)[j-1].timestamp;
92  if (std::abs(timediff) > 35){ // If the "next" (j-1) word is to early, stop the loop.
93  start = j;
94  break;
95  }
96  }
97 
98  for (unsigned int j = i ; j < buffer->GetSize()-1 ; ++j){
99  __builtin_prefetch(&buffer[j+1]);
100  __builtin_prefetch(&buffer[j+2]);
101  int64_t timediff = (*buffer)[j+1].timestamp - ClockStart;
102  if (std::abs(timediff) > 150){ // if the "next" word (j+1) is to late, stop the loop.
103  stop = j;
104  break;
105  } else if (std::abs(timediff) >= 5 && end < i){
106  end = j;
107  }
108  }
109 
110  if (end > i){
111  n_data = end;
112  curr_Buf = end+1;
113  } else {
114  n_data = stop;
115  curr_Buf = stop+1; // Where to continue the search for E strip words.
116  }
117  event.length = stop - start + 1;
118  event.words.reset(new word_t[stop-start+1]);
119  event.timestamp = timeOfEvent;
120 
121  // Setting the data words to the event structure.
122  for (int j = start ; j < stop + 1 ; ++j){
123  __builtin_prefetch(&event.words[j-start+1],1);
124  __builtin_prefetch(&buffer[j+1]);
125  event.words[j-start] = (*buffer)[j];
126  }
127 
128  return true;
129 }

131 bool Unpacker::UnpackOneEvent(Event &event, int &n_data)
132 {
133  for (unsigned int i = curr_Buf ; i < buffer->GetSize() ; ++i){
134  __builtin_prefetch(&buffer[i+1]);
135  // Hard coding of the addresses of E strips for telescope A for improved speed.
136  if ((*buffer)[i].address >= 384 && (*buffer)[i].address <= 399){
137  // "Trigger" time.
138  int64_t tEvent = (*buffer)[i].timestamp;
139  event.Reset();
140  event.trigTel = 'A'; // To notify what telescope was triggered.
141  return ExtractEvent(event, n_data, i, tEvent);
142  // Hard coding of the addresses of E strips for telescope B for improved speed.
143  } else if ((*buffer)[i].address >= 480 && (*buffer)[i].address <= 495){
144  // "Trigger" time.
145  int64_t tEvent = (*buffer)[i].timestamp;
146  event.Reset();
147  event.trigTel = 'B';
148  return ExtractEvent(event, n_data, i, tEvent);
149  }
150  }
151  n_data = buffer->GetSize() + 1;
152  return false; // End of buffer is reached.
153 }
