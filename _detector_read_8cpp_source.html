204 {

211  else if (comp == "UNKNOWN") return unknown;

435 bool Detectors::setAll(std::string filename)
436 {
437  std::ifstream input_file(filename.c_str());
438 
439  std::string line;
440 
441  bool status = true;
442 
443  while ( next_line(input_file, line) ){
444  if (line.size()==0 || line[0]=='#') continue;
445 
446  std::istringstream icmd(line.c_str());
447  std::string name, type, telescope;
448  uint16_t addr;
449  double theta, phi;
450  icmd >> addr;
451  icmd >> name;
452  icmd >> theta;
453  icmd >> phi;
454  icmd >> type;
455  icmd >> telescope;
456  if (name.empty()) status = false;
457  if (type.empty()) status = false;
458 
459  Add(addr, Dinfo(name, Dtype(type), telescope, theta, phi));
460  }
461 
462  input_file.close();
463  return status;
464 }

468 bool Detectors::next_line(std::istream& in, std::string& cmd_line)
469 {
470  cmd_line = "";
471  std::string line;
472  while ( getline(in, line) ){
473  size_t ls = line.size();
474  if ( ls == 0 ){
475  break;
476  } else if ( line[ls-1] != '\\' ) {
477  cmd_line += line;
478  break;
479  } else {
480  cmd_line += line.substr(0, ls-1);
481  }
482  }
483  return in || !cmd_line.empty();
484 }
