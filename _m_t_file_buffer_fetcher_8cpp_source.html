129 template<class T, int N, bool created>
130 T* RingBuffer<T,N,created>::Put(T* t)
131 {
132  if ( Full() )
133  return 0;
134  if ( created ) {
135  t = ring[head];
136  } else {
137  ring[head] = t;
138  }
139  head = (head+1)%N;
140  size += 1;
141  return t;
142 }

255 WordBuffer* PrefetchThread::ReadingBegins()
256 {
257  PThreadMutexLock lock( mutex );
258  while ( true ) {
259  if ( finished && readRing.Empty() )
260  return 0;
261  if ( readRing.Empty() ){
262  mutex.Wait( &cond_avail );
263  continue;
264  }
265  return readRing.Get();
266  }

269 void PrefetchThread::StartReading()
270 {
271  while ( !cancel && !finished ){
272  WordBuffer* buffer = 0;
273  { // Critical section
274  PThreadMutexLock lock( mutex );
275  while ( !cancel && !finished && writeRing.Full() )
276  mutex.Wait( &cond_full );
277  if ( cancel || finished )
278  return;
279 
280  // Claim a buffer, but not yet say it is filled.
281  buffer = writeRing.Put();
282  } // Unlock in 'lock' destructor
283 
284 
285  // Reading is time-consuming and should be performed while the
286  // lock is released
287  if ( reader->Read(buffer) <= 0)
288  finished = true;
289  { // Critical section
290  PThreadMutexLock lock( mutex );
291  if ( !finished ) {
292  // Mark the buffer as readable
293  readRing.Put(buffer);
294  } else {
295  writeRing.Get();
296  }
297  // Tell main thread that data is available
298  pthread_cond_signal( &cond_avail );
299  } // Unlock in 'lock' destructor.
300  }
301 }
