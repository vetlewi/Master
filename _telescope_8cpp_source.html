70 double Telescope::CalcAngle(const uint16_t &xaddress, const uint16_t &yaddress) const
71 {
72  int n = findVert(xaddress);
73  int m = findHori(yaddress);
74 
75  if (n < 0 || m < 0)
76  return 0;
77 
78  double xaxis = 6.0 - (n - 17./2.)*0.3;
79  double yaxis = 6.0 + (n - 17./2.)*0.3;
80  double zaxis = (m - 17./2.)*0.3;
81  return acos(yaxis/sqrt(xaxis*xaxis + yaxis*yaxis + zaxis*zaxis))*180./acos(-1);
82 }

163 bool Telescope::next_line(std::istream &in, std::string &cmd_line)
164 {
165  cmd_line = "";
166  std::string line;
167  while ( getline(in, line) ){
168  size_t ls = line.size();
169  if ( ls == 0 ){
170  break;
171  } else if ( line[ls - 1] != '\\' ){
172  cmd_line += line;
173  break;
174  } else {
175  cmd_line += line.substr(0, ls-1);
176  }
177  }
178  return in || !cmd_line.empty();
