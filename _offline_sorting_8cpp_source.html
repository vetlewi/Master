64 static std::string trim_whitespace( const std::string& text )
65 {
66  size_t start = text.find_first_not_of(" \t");
67  if ( start == std::string::npos )
68  return "";
69 
70  size_t end = text.find_last_not_of(" \t");
71  if ( end == std::string::npos || end<=start )
72  end = text.size();
73  return text.substr(start, end-start+1);
74 }

104 bool OfflineSorting::SortBuffer(const WordBuffer* buffer)
105 {
106  Event event;
107  unpacker.SetBuffer(buffer);
108  Unpacker::Status ustat = Unpacker::END;
109  while (leaveprog == 'n'){
110  ustat = unpacker.Next(event);
111  if ( ustat != Unpacker::OKAY )
112  break;
113  nWords += event.length;
114 #ifndef SINGLES
115  userSort.Sort(event);
116 #endif // SINGLES
117  nEvents += 1;
118  }
119  return ustat == Unpacker::END;
120 }

122 bool OfflineSorting::SortFile(const std::string filename, int buf_start, int buf_end)
123 {
124  // Open data file.
125  if (bufferFetcher->Open(filename, buf_start) != BufferFetcher::OKAY){
126  std::cerr << "Data: Could not open '" << filename << "'" << std::endl;
127  return false;
128  }
129 
130  int buffer_count = 0, bad_buffer_count = 0;
131  rateMeter.Reset();
132  nEvents = 0;
133  nWords = 0;
134 
135  // Loop over all buffers
136  for (int b = buf_start ; buf_end < 0 || b < buf_end ; ++b){
137  // Stop if Ctrl-C has been passed
138  if (leaveprog != 'n'){
139  break;
140  }
141 
142  // Fetch next buffer.
143  BufferFetcher::Status fstate;
144  const WordBuffer* buf = bufferFetcher->Next(fstate);
145  if ( fstate == BufferFetcher::END ){
146  break;
147  } else if ( fstate == BufferFetcher::ERROR) {
148  std::cerr << "Data: Error reading buffer " << b << " in file '" << filename << "'" << std::endl;
149  return false;
150  }
151 
152  // Sort buffer
153  buffer_count += 1;
154 #ifndef SINGLES
155  bool sort_ok = SortBuffer(buf);
156 #else
157  Event singles;
158  singles.length = buf->GetSize();
159  singles.words.reset(new word_t[buf->GetSize()]);
160  for ( int k = 0 ; k < buf->GetSize() ; k++)
161  singles.words[k] = (*buf)[k];
162  const bool sort_ok = userSort.SortSingles(singles);
163  singles.Reset();
164 #endif
165  if ( !sort_ok ){
166  bad_buffer_count += 1;
167  }
168 
169  const float bufs_per_sec = rateMeter.Rate();
170  if ( bufs_per_sec > 0 ){
171  if ( is_tty ){
172  std::cout << " " << std::flush << '\r' // Clear the line
173  << buffer_count << '/' << bad_buffer_count
174  << ' ' << bufs_per_sec << " bufs/s " << float(nEvents)/buffer_count << " Events/bufs " << float(nWords)/nEvents << " avr. nWords/event" << std::flush;
175  } else {
176  std::cout << '.' << std::flush;
177  }
178  }
179 
180  }
181 
182  // Print counter and rate at the end
183  std::cout << '\r' << buffer_count << '/' << bad_buffer_count
184  << ' ' << rateMeter.TotalRate() << " bufs/s " << float(nEvents)/buffer_count << " Events/bufs " << nEvents << " events sorted " << float(nWords)/nEvents << " avr. number of words/event" << std::endl;
185  return true;
186 
187 }

189 bool OfflineSorting::export_command(std::istream& icmd)
190 {
191  std::string tmp;
192  icmd >> tmp;
193  if (tmp == "root"){
194  icmd >> tmp;
195  std::string rootfile = trim_whitespace( tmp );
196  if (rootfile.empty()){
197  std::cerr << "Export root: Do not understand ROOT filename: '" << tmp << "'" << std::endl;
198  return false;
199  }
200  RootWriter::Write( userSort.GetHistograms(), rootfile );
201  std::cout << "Export as ROOT file into '" << tmp << "'" << std::endl;
202  userSort.GetHistograms().ResetAll();
203  return true;
204  } else if (tmp == "mama"){
205  icmd >> tmp;
206  std::string histname = trim_whitespace(tmp);
207  if ( histname.empty() ){
208  std::cerr << "Export mama: Do not understand histogram name '" << histname << "'" << std::endl;
209  return false;
210  }
211  Histogram1Dp h = userSort.GetHistograms().Find1D(histname);
212  Histogram2Dp m = userSort.GetHistograms().Find2D(histname);
213  if (!h && !m){
214  std::cerr << "Export mama: No histogram named '" << histname << "'." << std::endl;
215  return false;
216  }
217 
218  icmd >> tmp;
219  std::string mamafile = trim_whitespace(tmp);
220  if (mamafile.empty()){
221  std::cerr << "Export mama: Do not understand mama fiename '" << mamafile << "'" << std::endl;
222  return false;
223  }
224  std::ofstream mama_out( mamafile.c_str() );
225  if ( !mama_out ){
226  std::cerr << "Export mama: Problem opening '" << mamafile << "'." << std::endl;
227  return false;
228  }
229  std::cout << "Export '" << histname << "' as MAMA file into '" << mamafile << "'" << std::endl;
230  if ( h )
231  MamaWriter::Write(mama_out, h);
232  else if ( m )
233  MamaWriter::Write(mama_out, m);
234  if ( !mama_out ){
235  std::cerr << "Export mama: Problem writing '" << mamafile << "'." << std::endl;
236  return false;
237  }
238  mama_out.close();
239  return true;
240  }
241  return false;

317 bool OfflineSorting::next_commandline(std::istream& in, std::string& cmd_line)
318 {
319  cmd_line = "";
320  std::string line;
321 
322  while ( getline(in, line) ){
323  size_t ls = line.size();
324  if ( ls == 0 ){
325  break;
326  } else if ( line[ls-1] != '\\'){
327  cmd_line += line;

329  } else {

331  }
332  }
