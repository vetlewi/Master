196 bool UserSort::Sort(const Event &event)
197 {
198  nEvents += 1;
199  nWords += event.length;
200  _rando = drand48() - 0.5;
201 
202  // Reseting variables.
203  int ndEx=0, ndEy=0, ndE=0;
204  nL1=0, nL2=0;
205  nXA=0; nYA=0; nEA=0;
206  nXB=0; nYB=0; nEB=0;
207  for (int i = 0 ; i < 8 ; ++i)
208  nC[i] = 0;
209 
210  std::string telescope;
211  word_t dX[16], dY[16], dE[16];
212 
213  // Defining some variables used in loops. May
214  // give a slight increase in performance.
215  int64_t timediff;
216  double eSi, e_addback, e_bg, e_thisWord, eDop, eLaBr;
217 
218  bool g_in_coin = false;
219 
220  // Sorting out the data words of the different detectors.
221  for (int i = 0 ; i < event.length ; ++i){
222  __builtin_prefetch(&event.words[i+1]);
223  telescope = detectors->Find(event.words[i].address).getTelescope();
224  uint16_t address = event.words[i].address;
225  if (telescope[0] == 'C'){
226  int cnum = telescope[1] - '1';
227  if (nC[cnum] < 256)
228  clover[cnum][nC[cnum]] = event.words[i];
229  nC[cnum] += 1;
230  } else if (address == 520){ // Hardcode for speed.
231  if (nL1 < 256)
232  labr1[nL1] = event.words[i];
233  ++nL1;
234  } else if (address == 521){ // Hardcode for speed.
235  if (nL2 < 256)
236  labr2[nL2] = event.words[i];
237  ++nL2;
238  } else if (320 <= address && address <= 335){
239  if (nXA < 256)
240  dXA[nXA] = event.words[i];
241  ++nXA;
242  } else if (352 <= address && address <= 367){
243  if (nYA < 256)
244  dYA[nYA] = event.words[i];
245  ++nYA;
246  } else if (384 <= address && address <= 399){
247  if (nEA < 256)
248  dEA[nEA] = event.words[i];
249  ++nEA;
250  } else if (416 <= address && address <= 431){
251  if (nXB < 256)
252  dXB[nXB] = event.words[i];
253  nXB += 1;
254  } else if (448 <= address && address <= 463){
255  if (nYB < 256)
256  dYB[nYB] = event.words[i];
257  ++nYB;
258  } else if (480 <= address && address <= 495){
259  if (nEB < 256)
260  dEB[nEB] = event.words[i];
261  ++nEB;
262  }
263  }
264 
265  if (event.trigTel == 'A'){
266 
267  // Check if the rules for a "trigger" is not broken.
268 
269  for (int i = 0 ; i < nEB ; ++i){
270 
271  timediff = dEB[i].timestamp - event.timestamp;
272 
273  if (std::abs(timediff) < 5){
274  nDiscarded += 1;
275  return true;
276  }
277  }
278  for (int i = 0 ; i < nXA ; ++i){
279 
280  timediff = dXA[i].timestamp - event.timestamp;
281  eSi = GainEnergy(dXA[i].address, dXA[i].adcdata+_rando);
282 
283  m_E_t_X_A[dXA[i].address-320]->Fill(eSi, timediff*10);
284  if (std::abs(timediff) <= 6){
285  if (ndEx < 16)
286  dX[ndEx] = dXA[i];
287  ++ndEx;
288  }
289  }
290  for (int i = 0 ; i < nYA ; ++i){
291 
292  timediff = dYA[i].timestamp - event.timestamp;
293  eSi = GainEnergy(dYA[i].address, dYA[i].adcdata+_rando);
294 
295  m_E_t_Y_A[dYA[i].address-352]->Fill(eSi, timediff*10);
