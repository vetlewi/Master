26 static bool next_line(std::istream &in, std::string &cmd_line)
27 {
28  cmd_line = "";
29  std::string line;
30  while ( getline(in, line) ){
31  unsigned long ls = line.size();
32  if ( ls == 0 )
33  break;
34  else if ( line[ls - 1] != '\\' ){
35  cmd_line += line;
36  break;
37  } else {
38  cmd_line += line.substr(0, ls-1);
39  }
40  }
41  return in || !cmd_line.empty();
42 }

44 TimeGate::TimeGate(const std::string &filename)
