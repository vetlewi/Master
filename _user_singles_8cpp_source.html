108 bool UserSingles::SortSingles(const Event& event)
109 {
110 
111 
112  _rando = drand48() - 0.5;
113 
114 
115  std::string telescope;
116  double Energy;
117  for (int i = 0 ; i < event.length-1 ; ++i){
118 
119  telescope = detectors->Find(event.words[i].address).getTelescope();
120  Energy = GainEnergy(event.words[i].address, event.words[i].adcdata+_rando);
121  if (telescope[0] == 'C'){
122  h_Clover_crystal[telescope[1]-'1'][telescope[2] - 'A']->Fill(Energy);
123  h_Clover_crystal_raw[telescope[1]-'1'][telescope[2] - 'A']->Fill(event.words[i].adcdata);
124  if (i + 1 < event.length){
125  if (detectors->Find(event.words[i+1].address).getTelescope()[1] == telescope[1]){
126  Energy += GainEnergy(event.words[i+1].address, event.words[i+1].adcdata+_rando);
127  ++i;
128  }
129  }
130  if (doppler)
131  Energy = Dop(Energy, detectors->Find(event.words[i].address).getTheta(), detectors->Find(event.words[i].address).getPhi());
132  h_Clover_E[telescope[1] - '1']->Fill(Energy);
133  m_Gamma_E->Fill(Energy, telescope[1] - '1');
134  } else if (telescope[0] == 'L'){
135  if (doppler)
136  Energy = Dop(Energy, detectors->Find(event.words[i].address).getTheta(), detectors->Find(event.words[i].address).getPhi());
137  h_LaBr_E[telescope[1] - '1']->Fill(Energy);
138  h_LaBr_E_raw[telescope[1] - '1']->Fill(event.words[i].adcdata);
139  m_Gamma_E->Fill(Energy, telescope[1] - '1' + 8);
140  } else if (event.words[i].address >= 320 && event.words[i].address <= 335){
141  h_DSSD_A_dX[event.words[i].address - 320]->Fill(Energy);
142  h_DSSD_A_dX_raw[event.words[i].address - 320]->Fill(event.words[i].adcdata);
143  m_Particle_E->Fill(Energy, event.words[i].address - 320);
144  } else if (event.words[i].address >= 352 && event.words[i].address <= 367){
145  h_DSSD_A_dY[event.words[i].address - 352]->Fill(Energy);
146  h_DSSD_A_dY_raw[event.words[i].address - 352]->Fill(event.words[i].adcdata);
147  m_Particle_E->Fill(Energy, event.words[i].address - 352 + 16);
148  } else if (event.words[i].address >= 384 && event.words[i].address <= 399){
149  h_DSSD_A_E[event.words[i].address - 384]->Fill(Energy);
